    uu.forEach({ a: "a", b: 1 }, function(v, i) {
      uu.log(i + ":" + v); // "a:a", "b:1"
    });
    uu.forEach(["a", 1], function(v, i) {
      uu.log(i + ":" + v); // "0:a", "1:1"
    });
    uu.forEach(uu.tag("*"), function(v, i) {
      uu.log(i + ":" + v); // "0:[object HTMLHtmlElement]", "1:[object HTMLHeadElement]", ...
    });
