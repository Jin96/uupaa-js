function matrix() {
    draw(document.getElementById('slcanvas'), 1);
    draw(document.getElementById('vmlcanvas'), 1);
    draw(document.getElementById('flashcanvas'), 1);
}
function draw(canvas, matrix){
    if (!canvas.getContext){ return; }

    var ctx = canvas.getContext('2d');

    if (matrix) {
        ctx.clear();
        ctx.translate(-10, 5);
        ctx.scale(1.2, 0.8);
        ctx.rotate(5 * Math.PI / 180);
    }
    var img = new Image();
    img.onload = function(){
        ctx.scale(1, 0.3);
        ctx.translate(20, 20);
        ctx.rotate(3 * Math.PI / 180);
        ctx.beginPath();
        ctx.arc(80, 100, 80, 0, Math.PI * 2, 1);
    //    ctx.rect(100, 100, 100, 100);
        ctx.clip();
        ctx.strokeStyle = "red";
        ctx.lineWidth = 3;
        ctx.drawImage(img, 0, 0);
        ctx.stroke();
    }
    img.src = 'img/rhino.jpg?' + new Date().getTime();
}
