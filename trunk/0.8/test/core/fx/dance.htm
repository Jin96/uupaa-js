function action() {
    uu("div").each(function(node, i) {
        uu.fx(node, 1000, { x: i * 100, y: i * 100, chain: 1 });
    }).fx(1000, { x: 0, y: 0, chain: 1 }).each(function(node, i) {
        uu.fx(node, 1000, { x: 400, y: 400, chain: 1 });
        uu.fx(node, 1000, { x: i * 100, y: i * 100, chain: 1 });
    }).fx(1000, { x: 0, y: 0, reverse: 1 });
}
