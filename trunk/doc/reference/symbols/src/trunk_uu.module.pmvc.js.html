  6 (function() { var /* uud = document, */ uuw = window, uu = uuw.uu;
  7 
  8 /** <b>Pluggable MVC</b>
  9  *
 10  * @class
 11  */
 12 uu.module.pmvc = function() {
 13 };
 14 
 15 /** <b>Pluggable MVC - Application Model</b>
 16  *
 17  * @class
 18  */
 19 uu.module.pmvc.applicationModel = uu.basicClass();
 20 
 21 /** <b>Pluggable MVC - Domain Model</b>
 22  *
 23  * @class
 24  */
 25 uu.module.pmvc.domainModel = uu.basicClass();
 26 
 27 /** <b>Pluggable MVC - controller</b>
 28  *
 29  * @class
 30  */
 31 uu.module.pmvc.controller = uu.basicClass();
 32 
 33 /** <b>Pluggable MVC - View</b>
 34  *
 35  * @class
 36  */
 37 uu.module.pmvc.view = uu.basicClass();
 38 
 39 uu.forEach({
 40   /** @scope uu.module.pmvc.controller */
 41 
 42   /** <b>初期化</b>
 43    *
 44    * @param string id       - ユニークなID(メッセージ配達用のアドレス)を指定します。
 45    */
 46   construct: function(id) {
 47     this.id = id;
 48     this.catalog = { ping: "_ping" }; // msg-name: function-name
 49     uu.msgpump.set(this.id, this);
 50     this.activate();
 51   },
 52   /** <b>メッセージとハンドラの登録</b>
 53    *
 54    * @param hash handler - hash { msg, function() { ... }, ... } を指定します。
 55    *                       msgで指定したメッセージが到着すると、
 56    *                       functionで指定した関数が呼ばれます。
 57    */
 58   regist: function(handler) {
 59     var me = this;
 60     uu.forEach(handler, function(fn, msg) {
 61       me.catalog[msg] = msg; // 実名で登録
 62       me[msg] = fn;
 63     });
 64   },
 65   /** <b>エリアスメッセージカタログの登録</b>
 66    *
 67    * メッセージの別名を登録します。
 68    *
 69    * @param hash catalog - エリアス(別名)とメッセージ(実体)のカタログです。
 70    *                       hash { alias, msg, ... } の形で指定します。
 71    *                       aliasで指定したメッセージが到着すると、
 72    *                       msgとリンクしているハンドラが呼ばれます。
 73    * @throws TypeError "uu.module.pmvc::registArias(catalog) no unsubstantial" 実体が無い
 74    */
 75   registArias: function(catalog) {
 76     var me = this;
 77     uu.forEach(catalog, function(msg, alias) {
 78       if (!(msg in me.catalog)) { // 実体が無い
 79         throw TypeError("uu.module.pmvc::registArias(catalog) no unsubstantial");
 80       }
 81       me.catalog[alias] = msg; // { msg-name(alias): function-name }
 82     });
 83   },
 84   /** <b>活性化</b>
 85    * メッセージの受け取りを開始します。
 86    */
 87   activate: function() {
 88     this.permit = 0x1;
 89   },
 90   /** <b>不活性化</b>
 91    * メッセージの受け取りを停止します。
 92    */
 93   deactivate: function() {
 94     this.permit = 0x0;
 95   },
 96   /** <b>メッセージの受信とハンドラの呼び出し</b>
 97    *
 98    * @param string msg   - メッセージの名前を指定します。
 99    * @param mix [param1] - 引数を指定します。
100    * @param mix [param2] - 引数を指定します。
101    * @return bool - 呼び出し成功でtrue, 失敗でfalseを返します。
102    */
103   procedure: function(msg, param1, param2) {
104     if (!this._hook(msg, param1, param2) || !this.permit) { return false; }
105     if (!(msg in this.catalog)) { return this._unknown(msg, param1, param2); }
106     return this[this.catalog[msg]].call(this, msg, param1, param2);
107   },
108   _ping:    function(msg, param1, param2) { alert(this.id + " - alive"); },
109   _hook:    function(msg, param1, param2) { return true; }, // falseでルーティング終了
110   _unknown: function(msg, param1, param2) { return true; }  // 未知のメッセージ用ハンドラ
111 }, function(v, p) {
112   uu.module.pmvc.applicationModel.prototype[p] = v;
113   uu.module.pmvc.domainModel.prototype[p] = v;
114   uu.module.pmvc.controller.prototype[p] = v;
115   uu.module.pmvc.view.prototype[p] = v;
116 });
117 
118 /* activate primary instance */
119 uu.module.pmvc.activate = function() {
120   uu.app    = new uu.module.pmvc.applicationModel("A");
121   uu.domain = new uu.module.pmvc.domainModel("D");
122   uu.view   = new uu.module.pmvc.view("V");
123   uu.ctrl   = new uu.module.pmvc.controller("C");
124 };
125 
126 })(); // end (function())()
