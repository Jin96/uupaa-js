 10 /** <b>キャンバス要素を動的に生成する(create dynamic canvas)</b>
 11  *
 12  * キャンバス要素を動的に生成します。
 13  * キャンバスはposition: absolute, display: noneの状態で生成されます。
 14  *
 15  * @param Number  width             - canvasの幅をpx単位で指定します。デフォルトは300です。
 16  * @param Number  height            - canvasの高さをpx単位で指定します。デフォルトは300です。
 17  * @param Hash    [style]           - スタイルの指定です。
 18  * @param Number  [style.left]      - canvasのx座標をpx単位で指定します。デフォルトは0です。
 19  * @param Number  [style.top]       - canvasのy座標をpx単位で指定します。デフォルトは0です。
 20  * @param Number  [style.display]   - canvasの表示方法(display)を指定します。デフォルトは"none"です。
 21  * @param Number  [style.position]  - canvasの配置(position)を指定します。デフォルトは"absolute"です。
 22  * @return Element                  - 生成したcanvas要素を返します。
 23  */
 24 uu.canvas.create = function(width /* = 300 */, height /* = 300 */, style /* = { x: 0, y: 0, w: 300, h: 300, display: "none", position: "absolute" } */) {
 25   style = uu.mix.param(style || {}, { left: 0, top: 0, display: "none", position: "absolute" });
 26   var e = uud.createElement("canvas");
 27   uu.mix(e, { width: width || 300, height: height || 300 });
 28   uu.css.set(e, style);
 29   return e;
 30 };
 31 
 32 /** <b>コンテキストの取得</b>
 33  *
 34  * コンテキストを取得します。excanvas.jsで動的に生成したcanvasの初期化も行います。
 35  *
 36  * @param  element  elm    - canvasオブジェクトを指定します。
 37  * @param  string   type   - コンテキストタイプを指定します。デフォルトは"2d"です。
 38  * @return context           コンテキストを返します。
 39  */
 40 uu.canvas.context = function(elm, type /* = "2d" */) {
 41   type = type || "2d";
 42   return (!uu.ua.ie) ? elm.getContext(type)
 43                      : G_vmlCanvasManager.initElement(elm).getContext(type);
 44 };
 45 
 46 /** <b>グリッドを描画</b>
 47  *
 48  * グリッドを描画します。
 49  *
 50  * <pre class="eg">
 51  *  var canvas = document.getElementById("hoge");
 52  *  var ctx = canvas.getContext("2d");
 53  *  uu.canvas.drawGrid(ctx, function() {
 54  *    ;
 55  *  });
 56  * </pre>
 57  *
 58  * @param context2d ctx   - 2d contextを指定します。
 59  * @param Function  fn    - 描画完了で呼び出す関数を指定します。デフォルトはundefinedです。
 60  * @param Number    size  - グリッドサイズを指定します。単位はpxです。デフォルトは64です。
 61  * @param String    color - 色を指定します。デフォルトは"gray"です。
 62  */
 63 uu.canvas.drawGrid = function(ctx, fn /* = undefined */, size /* = 64 */, color /* = "gray" */) {
 64   fn = fn || uu.mute;
 65   size = size || 64;
 66   color = color || "gray";
 67   var x, y, url = uu.config.imagePath + "uu.module.canvas.grid.gif";
 68 
 69   uu.module.image.preload(url, function(code, _url, _img) {
 70     if (code && !uu.ua.ie) {
 71       ctx.beginPath();
 72       ctx.fillStyle = ctx.createPattern(_img, "");
 73       ctx.rect(0, 0, ctx.canvas.width, ctx.canvas.height);
 74       ctx.fill();
 75     };
 76     ctx.beginPath();
 77     ctx.strokeStyle = color;
 78     for (y = size; y < ctx.canvas.height; y += size) {
 79       ctx.moveTo(0, y);
 80       ctx.lineTo(ctx.canvas.width, y);
 81     }
 82     for (x = size; x < ctx.canvas.width; x += size) {
 83       ctx.moveTo(x, 0);
 84       ctx.lineTo(x, ctx.canvas.height);
 85     }
 86     ctx.stroke();
 87     fn();
 88   });
 89 };

 91 
 92 })(); // end (function())()
