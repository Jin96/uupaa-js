 47 /** <b>要素に関する情報を取得</b>
 48  *
 49 <pre>
 50 offsetLeft
 51 ├───┤
 52 ┏━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 53 ┃
 54 ┃      ┌  ─  ─  ─  ─  ─  ─  ─  ─  ─  ─  ─  ┐
 55 ┃                      ↑margin-top                      
 56 ┃      │              ↓                              │
 57 ┃        ┏━━━━━━━━━━━━━━━━━━━━━┓  
 58 ┃      │┃■■■■■■↑border-top■■■■■■■■■┃│
 59 ┃        ┃■■■■■■↓■■■■■■■■■■■■■■┃  
 60 ┃      │┃■■┏━━━━━━━━━━━━━━━┓■■┃│
 61 ┃        ┃■■┃      ↑padding-top           ┃■■┃  
 62 ┃      │┃■■┃      ↓                      ┃■■┃│
 63 ┃        ┃■■┃  ┌─Contents─Area───┐  ┃■■┃  
 64 ┃      │┃■■┃  │                      │  ┃■■┃│
 65 ┃        ┃■■┃  │                      │  ┃■■┃  
 66 ┃      │┃■■┃  │                      │  ┃■■┃│
 67 ┃        ┃■■┃  │                      │  ┃■■┃  
 68 ┃      │┃■■┃  │                      │  ┃■■┃│
 69 ┃        ┃■■┃  └───────────┘  ┃■■┃  
 70 ┃      │┃■■┃                              ┃■■┃│
 71 ┃        ┃■■┗━━━━━━━━━━━━━━━┛■■┃  
 72 ┃      │┃■■■■■■■■■■■■■■■■■■■■■┃│
 73 ┃        ┃■■■■■■■■■■■■■■■■■■■■■┃  
 74 ┃      │┗━━━━━━━━━━━━━━━━━━━━━┛│
 75 ┃      └  ─  ─  ─  ─  ─  ─  ─  ─  ─  ─  ─  ┘
 76 ┃        ├───────offsetWidth-────────┤
 77 ┃      
 78 ┃              ├────clientWidth-─────┤
 79 ┃      
 80 ┃        ├──┤
 81 ┃       clientLeft (Firefox3, 左のマージンとパディングを含まないボーダー幅のピクセル数)
 82 ┃  
 83 </pre>
 84  *
 85  *  <dl>
 86  *    <dt>x</dt><dd>画面左上からのオフセット幅(スクロール量を含む) - offsetLeftの累計 - ウインドウの原点からの絶対座標</dd>
 87  *    <dt>y</dt><dd>画面左上からのオフセット高さ(スクロール量を含む) - offsetTopの累計 - ウインドウの原点からの絶対座標</dd>
 88  *    <dt>w</dt><dd>要素の幅(オフセット幅, 左右のボーダーを含む) - offsetWidth</dd>
 89  *    <dt>h</dt><dd>要素の高さ(オフセット高さ, 上下のボーダーを含む) - offsetHeight</dd>
 90  *    <dt>cw</dt><dd>要素の幅(オフセット幅, 左右のボーダーを含まない) - clientWidth</dd>
 91  *    <dt>ch</dt><dd>要素の高さ(オフセット高さ, 上下のボーダーを含まない) - clientHeight</dd>
 92  *  </dl>
 93  *
 94  * @param   element   elm - 要素を指定します。
 95  * @return  rect          - { x, y, w, h } を返します。<br />
 96  *                          x, y: 画面左上からの絶対座標。<br />
 97  *                          w, h: 要素の幅と高さ
 98  * @namespace
 99  */
100 uu.ui.element = function(elm) {
101   var e = elm, x = 0, y = 0;
102   while (e) {
103     x += e.offsetLeft || 0;
104     y += e.offsetTop  || 0;
105     e = e.offsetParent;
106   }
107   return { x: x, y: y,
108            w: elm.offsetWidth, h: elm.offsetHeight,
109            cw: elm.clientWidth, ch: elm.clientHeight };
110 };
111 if ("getBoxObjectFor" in uud) { // for Firefox
112   uu.ui.element = function(elm) {
113     var b = uud.getBoxObjectFor(elm), inn = uu.ui.inner();
114     return { x: inn.sw + b.x, y: inn.sh + b.y,
115              w: b.width, h: b.height,
116              cw: elm.clientWidth, ch: elm.clientHeight };
117   };
118 } else if ("getBoundingClientRect" in uud) { // for IE, +Firefox3.0, +Opera9.5β2
119   uu.ui.element = function(elm) {
120     var b = elm.getBoundingClientRect(), inn = uu.ui.inner();
121     return { x: inn.sw + b.left - 2, y: inn.sh + b.top - 2, // Operaで-2する必要があるかは要確認
122              w: elm.offsetWidth, h: elm.offsetHeight,
123              cw: elm.clientWidth, ch: elm.clientHeight };
124   };
125 }

127 /** <b>最寄の基準点からの累計オフセットと要素の大きさを取得</b>
128  *
129  * 最寄の基準点からの累計オフセットと要素の大きさを取得します。
130  * 最寄の基準点とは、position absoluteやrelativeが指定された要素のことです。
131  *
132  * @param   element   elm - 要素を指定します。
133  * @return  rect          - { x, y, w, h } を返します。<br />
134  *                          x, y: 最寄の基準点からの累計オフセット値。<br />
135  *                          w, h: 要素の幅と高さ
136  */
137 uu.ui.element.offsetParent = function(elm) {
138   var e = elm, x = 0, y = 0, ss;
139   while (e) {
140     x += e.offsetLeft || 0;
141     y += e.offsetTop  || 0;
142     e = e.offsetParent;
143     if (e) {
144       ss = uu.css.get(e, "position");
145       if (ss === "relative" || ss === "absolute") { break; }
146     }
147   }
148   return { x: x, y: y,
149            w: elm.offsetWidth, h: elm.offsetHeight,
150            cw: elm.clientWidth, ch: elm.clientHeight };
151 };

153 /** <b>絶対座標化</b>
154  *
155  * 要素の現在位置を保持したまま絶対座標化します。
156  *
157  * @param   element   elm     - 要素を指定します。
158  * @param   hash      [style] - 追加するスタイルを { 名前: 値, ... }の形式で指定します。<br />
159  *                              デフォルトはundefinedです。
160  * @return  element           - elmを返します。
161  */
162 uu.ui.element.toAbsolute = function(elm, style /* = undefined */) {
163   var rect = uu.ui.element(elm);
164   style = uu.mix.param(style || {}, { position: "absolute", left: rect.x + "px", top: rect.y + "px" });
165   uu.css.set(elm, style);
166   return elm;
167 };

173  *                              デフォルトはundefinedです。
174  * @return  element           - elmを返します。

183 /** <b>座標が矩形内にあればtrue</b>
184  *
185  * 座標(マウスカーソル等)が矩形内に含まれている場合にtrueを返します。
186  *
187  * <pre>
188  *  rect━┯━━┓
189  *    ┃  │    ┃
190  *    ┠─pos   ┃
191  *    ┃        ┃
192  *    ┗━━━━┛
193  * </pre>
194  *
195  * @param   rect    rect - { x, y, w, h } 要素の座標と大きさを指定します。
196  * @param   pos     pos  - { x, y } 点を指定します。
197  * @return  bool         - rect上にposがあればtrueを返します。
198  */
199 uu.ui.inRect = function(rect, pos) {
200   var r = rect, p = pos;
201   return (p.x > r.x && p.x < r.x + r.w) && (p.y > r.y && p.y < r.y + r.h);
202 };

280 /** <b>マウスの座標と要素のオフセット値を取得</b>
281  *
282  * マウスによるドラッグ操作用のコンビニエンス関数です。<br />
283  *
284  * 以下の状態を含むhashを返します。<br />
285  * x, y: ページ座標(絶対座標)と、マウスカーソルが乗っている要素の原点(左上)とのオフセット値
286  *
287  * @return hash - { x, y } を返します。
288  */
289 uu.event.mouse.offsetElement = function(evt, elm) {
290   var mpos = uu.event.mouse.pos(evt);
291   return { x: mpos.x - parseInt(elm.style.left),
292            y: mpos.y - parseInt(elm.style.top) };
293 };

295 /** <b>マウスクリック, ホイールの状態を取得</b>
296  *
297  * 以下の状態を含むhashを返します。<br />
298  * left:  左クリックでtrue<br />
299  * mid:   中クリックでtrue<br />
300  * right: 右クリックでtrue<br />
301  * click: クリック数, シングルクリックで1, ダブルクリックで2, トリプルクリックで3<br />
302  * wheel: マウスホイールを上に回転させると-1を、下に回転させると1を返します<br />
303  *
304  * @param   event   evt - イベントオブジェクト
305  * @return  hash        - { left, mid, right, click, wheel } を返します。
306  */
307 uu.event.mouse.state = function(evt) {
308   var rv = { left: 0, mid: 0, right: 0, click: 0, wheel: 0 };
309   if (uu.ua.gecko || evt.which) {
310     switch (evt.which) {
311       case  1: rv.left  = 1; break;
312       case  2: rv.mid   = 1; break;
313       case  3: rv.right = 1; break;
314     }
315     rv.click = evt.detail & 0x03; // クリック回数(0～3)
316   } else if (uu.ua.ie || evt.button) {
317     if (evt.button & 0x1) { rv.left = 1;  }
318     if (evt.button & 0x4) { rv.mid = 1;   }
319     if (evt.button & 0x2) { rv.right = 1; }
320   }
321   // wheelDelta: Safari, IE
322   // detail: Firefox
323   if (evt.wheelDelta || evt.detail) {
324     rv.wheel = parseInt(evt.detail ? (evt.detail / 3) : (evt.wheelDelta / -120));
325   }
326   return rv;
327 };
