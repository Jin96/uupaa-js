  6 (function() { var /* uud = document, */ uuw = window, uu = uuw.uu;
  7 
  8 /** <b>Polygon</b>
  9  *
 10  * スクェアポリゴンを描画します。
 11  *
 12  * @class
 13  */
 14 uu.module.polygon = uu.basicClass();
 15 uu.module.polygon.prototype = {
 16   /** <b>uu.module.polygon.construct - 初期化</b>
 17    *
 18    * @param hash    [param]       - パラメタを指定します。
 19    * @param number  [param.fps]   - 1秒間に表示するフレーム数の指定です。
 20    *                                フレーム数を多くすると滑らかになりますが負荷も増えます。
 21    *                                25を指定すると約40msで1回描画します。
 22    *                                デフォルトは50です(20msで1回描画)。
 23    */
 24   construct: function(param /* = {} */) {
 25     this.param = uu.mix.param(param || {}, { fps: 50 });
 26     this.fps = 1000 / param.fps;
 27     this.cube = [];
 28     this.tmid = -1;
 29   },
 30   /** <b>uu.module.polygon.add - ポリゴンの追加</b>
 31    *
 32    * @param hash    [param]           - パラメタの指定です。
 33    * @param number  [param.highLight] - ハイライトの強さを0x0～0xffで指定します。デフォルトは0x40です。
 34    * @param number  [param.color]     - ポリゴンの色を指定します。0xff0000で赤, 0x00ff00で緑, 0x0000ffで青になります。デフォルトは0x000000(黒)です。
 35    * @param number  [param.opacity]   - 不透明度を0.0～1.0の値で指定します。デフォルトは1.0(完全な不透明)です。
 36    * @param number  [param.x]         - ポリゴンのx座標を指定します。デフォルトは0です。
 37    * @param number  [param.y]         - ポリゴンのy座標を指定します。デフォルトは0です。
 38    * @param number  [param.zoom]      - ポリゴンの拡大倍率を数値で指定します。デフォルトは600です。
 39    * @param number  [param.phi]       - 縦軸の回転速度を指定します。デフォルトは0.03141592653589793(Math.PI / 100)です。
 40    * @param number  [param.theta]     - 横軸の回転速度を指定します。デフォルトは0.1570796326794897(Math.PI / 80)です。
 41    */
 42   add: function(param /* = {} */) {
 43     param = uu.mix.param(param || {}, { highLight: 0x40, color: 0x000000, opacity: 1.0, x: 0, y: 0, zoom: 600, 
 44                                         phi: Math.PI / 100, theta: Math.PI / 80 });
 45     var data = [[], [], [], [], [], []], 
 46         i = 0, v1, v2;
 47     for (; i < 5; ++i) {
 48       v1 = (!i) ? 0 : Math.SQRT2 * Math.cos((0.5 * i - 0.25) * Math.PI);
 49       v2 = (!i) ? 0 : Math.SQRT2 * Math.sin((0.5 * i - 0.25) * Math.PI);
 50       data[0].push([ v1,  v2,   1]);
 51       data[1].push([  1,  v1,  v2]);
 52       data[2].push([ v2,   1,  v1]);
 53       data[3].push([-v1, -v2,  -1]);
 54       data[4].push([ -1, -v1, -v2]);
 55       data[5].push([-v2,  -1, -v1]);
 56     }
 57     this.cube.push(uu.mix(param, { _data: data, _theta: 0.5, _phi: 0.5 }));
 58   },
 59   /** <b>uu.module.polygon.draw - 描画</b>
 60    *
 61    * @param context2d ctx   - 描画先の2dコンテキストを指定します。
 62    * @param bool      [run] - 連続描画する場合にtrueを指定します。falseを指定すると1度だけ描画します。
 63    *                          デフォルトはfalseです。
 64    */
 65   draw: function(ctx, run /* = false */) {
 66     if (this.tmid !== -1) { return; }
 67     var me = this;
 68     if (run || false) {
 69       this.tmid = uuw.setInterval(function() {
 70         ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
 71         me.cube.forEach(function(v) {
 72           v._phi += v.phi;
 73           v._theta += v.theta;
 74           me._drawPolygon(ctx, v);
 75         });
 76       }, this.fps);
 77     } else {
 78       ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
 79       me.cube.forEach(function(v) {
 80         me._drawPolygon(ctx, v);
 81       });
 82     }
 83   },
 84   _drawPolygon: function(ctx, param) {
 85 
 86     var sinP = Math.sin(param._phi),   cosP = Math.cos(param._phi),
 87         sinT = Math.sin(param._theta), cosT = Math.cos(param._theta),
 88         d = param._data,
 89         // vector data
 90         vX = [-sinP,         cosP,            0],
 91         vY = [-cosT * cosP, -cosT * sinP,  sinT],
 92         vZ = [-sinT * cosP, -sinT * sinP, -cosT],
 93         info = [],
 94         x, y, z, i, j, light, surface; // 2D bitmap surface
 95 
 96     for (i = 0; i < d.length; ++i) {
 97       surface = [0, -(vZ[0] * d[i][0][0] +
 98                       vZ[1] * d[i][0][1] +
 99                       vZ[2] * d[i][0][2])];
100       for (j = 1; j < d[i].length; ++j) {
101         z = vZ[0] * d[i][j][0] +
102             vZ[1] * d[i][j][1] +
103             vZ[2] * d[i][j][2];
104         surface.push([vX[0] * d[i][j][0] +
105                       vX[1] * d[i][j][1] +
106                       vX[2] * d[i][j][2],
107                       vY[0] * d[i][j][0] +
108                       vY[1] * d[i][j][1] +
109                       vY[2] * d[i][j][2], z]);
110         surface[0] += z;
111       }
112       info.push(surface);
113     }
114     info.sort(this._sort);
115 
116     for (i = 0; i < info.length; ++i) {
117       info[i].shift();
118       light = info[i].shift();
119       if (light >= 0) {
120         for (j = 0; j < info[i].length; ++j) {
121           x = param.zoom * info[i][j][0] / (10 + info[i][j][2]);
122           y = param.zoom * info[i][j][1] / (10 + info[i][j][2]);
123           if (!j) {
124             ctx.beginPath();
125             ctx.moveTo(param.x + x, param.y + -y);
126             ctx.fillStyle = this._rgba(param.color, parseInt(light * param.highLight), param.opacity);
127           } else {
128             ctx.lineTo(param.x + x, param.y + -y);
129           }
130         }
131         ctx.closePath();
132         ctx.fill();
133       }
134     }
135   },
136   _sort: function(a, b) {
137     if (a[0] === b[0]) { return 0; }
138     return a[0] < b[0] ? 1: -1;
139   },
140   _rgba: function(color, bright, opacity) {
141     var rv = [Math.min(((color >> 16) & 0xff) + bright, 255),
142               Math.min(((color >> 8) & 0xff) + bright, 255),
143               Math.min((color & 0xff) + bright, 255), opacity];
144     return "rgba(" + rv.join(",") + ")";
145   }
146 };
147 
148 })(); // end (function())()
