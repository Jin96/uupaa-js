 11 /** <b>ブロック要素をドラッグで自由に移動可能にする(Free Drag and Drop)</b>
 12  *
 13  * @class
 14  */
 15 uu.module.drag.free = uu.basicClass();
 16 uu.module.drag.free.prototype = {
 17   /** <b>初期化</b>
 18    *
 19    * @param element   elm               - ドラッグする要素を指定します。
 20    * @param hash      [param]           - パラメタの指定です。
 21    * @param bool      [param.ghost]     - ゴーストエフェクト(fade)を使用する場合にtrueにします。デフォルトはtrueです。
 22    * @param string    [param.msgto]     - メッセージの受取先を指定します。""を指定するとメッセージを送信しません。
 23    *                                    "broadcast"を指定するとブロードキャストになります。デフォルトは""です。
 24    * @param function  [param.msgFilter] - 送信するメッセージを絞り込むフィルターを指定します。
 25    *                                      msgFilter(メッセージ名) の形で呼ばれ、falseに評価される値を返すとメッセージは送信されません。
 26    *                                      デフォルトはundefinedです。
 27    * @param bool      [param.resize]    - ホイールでリサイズする場合にtrueにします。<br />
 28    *                                      デフォルトは、trueです。
 29    */
 30   construct: function(elm, param /* = {} */) {
 31     this.elm = elm;
 32     this.param = uu.mix.param(param || {}, {
 33       ghost: true, cursor: "move",
 34       opacity: uu.css.get.opacity(this.elm),
 35       msgto: "", msgFilter: uu.echo, resize: true,
 36       _x: 0, _y: 0, _dragging: false
 37     });
 38 
 39     uu.ui.element.toAbsolute(this.elm);
 40     this.elm.style.cursor = this.param.cursor;
 41 
 42     // ghost effect. 見えない状態から本来の不透明度に戻す
 43     this.param.ghost && uu.effect.fadein(this.elm, { begin: 0, end: 1.0 });
 44 
 45     // z-index初期化
 46     this.zindexer = new uu.module.drag.zindexer();
 47     this.zindexer.set(this.elm.id);
 48 
 49     // EventHandler
 50     this.hr = uu.event.handler(this);
 51     uu.event.set(this.elm, "mousedown,mousewheel", this.hr);
 52   },
 53   /** <b>後処理</b> */
 54   destruct: function() {
 55     try {
 56       this.zindexer.unset(this.elm.id);
 57       uu.event.unset(this.elm, "mousedown,mousewheel", this.hr);
 58     } catch (e) {}
 59   },
 60   /** <b>イベントハンドラ</b> */
 61   handleEvent: function(evt) {
 62     var type = uu.event.type(evt.type);
 63     uu.event.stop(evt); // イベントバブルの停止(+テキストの選択を抑止)
 64     switch (type) {
 65     case "mousedown": uu.event.set(uu.ua.ie ? this.elm : uud, "mousemove,mouseup", this.hr, true); break;
 66     case "mouseup": uu.event.unset(uu.ua.ie ? this.elm : uud, "mousemove,mouseup", this.hr, true); break;
 67     }
 68     switch (type) {
 69     case "mousedown":
 70     case "mousemove":
 71     case "mouseup":
 72     case "mousewheel":
 73       this[type](evt);
 74       this.param.msgto && uu.msgpump && this.param.msgFilter(type) &&
 75         uu.msgpump.post(this.param.msgto, type, { sender: "uu.module.drag.free", element: this.elm });
 76       break;
 77     }
 78   },
 79   mousedown: function(evt) {
 80     var mpos = uu.event.mouse.pos(evt);
 81     // ドラッグ開始時のオフセット座標(マウス座標の絶対値 - ドラッグターゲットの原点座標)と不透明度を保存する
 82     this.param._x = mpos.x - parseInt(this.elm.style.left);
 83     this.param._y = mpos.y - parseInt(this.elm.style.top);
 84     this.dragging = true;
 85     this.zindexer.beginDrag(this.elm.id); // z-indexの更新
 86     this.param.ghost && uu.css.set.opacity(this.elm, 0.3); // 不透明度の設定
 87   },
 88   mousemove: function(evt) {
 89     if (!this.dragging) { return; }
 90 
 91     var mpos = uu.event.mouse.pos(evt);
 92     this.elm.style.left = parseInt(mpos.x - this.param._x) + "px";
 93     this.elm.style.top  = parseInt(mpos.y - this.param._y) + "px";
 94   },
 95   mouseup: function(evt) {
 96     if (!this.dragging) { return; }
 97     this.dragging = false;
 98     this.param.ghost && uu.effect.fadein(this.elm, { speed: "quick", end: 1.0 }); // 不透明度を戻す
 99     this.zindexer.endDrag(this.elm.id); // z-indexを戻す
100   },
101   mousewheel: function(evt) {
102     if (!this.param.resize) { return; }
103     var ss = uu.css.get(this.elm), wh = uu.event.mouse.state(evt).wheel * 2;
104     uu.css.set(this.elm, { width:  (parseInt(ss.width)  + (wh * 2)) + "px",
105                            height: (parseInt(ss.height) + (wh * 2)) + "px",
106                            top:    (parseInt(ss.top)    - wh) + "px",
107                            left:   (parseInt(ss.left)   - wh) + "px" });
108   }
109 };

111 /** <b>Limited Drag and Drop</b>
112  *
113  * idとclass="draggable"を持つdiv要素がドラッグ移動可能となり、
114  * class="droppable"を持つdiv要素にドロップ可能になります。
115  * droppable以外の場所でドロップしようとすると、元の場所に戻します。
116  *
117  * @class
118  */
119 uu.module.drag.limited = uu.basicClass();
120 uu.module.drag.limited.prototype = {
121   /** <b>ドラッグ可能要素の列挙とイベントハンドラのアサイン</b>
122    *
123    * @param hash      [param]       - パラメタの指定です。
124    * @param bool      [param.ghost] - ゴーストエフェクト(fade)を使用する場合にtrueにします。デフォルトはtrueです。
125    * @param string    [param.msgto] - メッセージの受取先を指定します。""を指定するとメッセージを送信しません。
126    *                                  "broadcast"を指定するとブロードキャストになります。デフォルトは""です。
127    * @param function  [param.msgFilter] - 送信するメッセージを絞り込むフィルターを指定します。
128    *                                      msgFilter(メッセージ名) の形で呼ばれ、falseに評価される値を返すとメッセージは送信されません。
129    *                                      デフォルトはundefinedです。
130    * @param bool      [param.resize]    - ホイールでリサイズする場合にtrueにします。<br />
131    *                                      デフォルトは、trueです。
132    * @param bool      [param.dropAllowColor] - ドロップ可能な要素の背景色の指定です。デフォルトは"bisque"です。
133    */
134   construct: function(param /* = {} */) {
135     var me = this;
136     this.elm = null; // drag target
137     this.param = uu.mix.param(param || {}, {
138       ghost: true, cursor: "move",
139       msgto: "", msgFilter: uu.echo, resize: true,
140       dropAllowColor: "bisque",
141       _x: 0, _y: 0
142     });
143 
144     this.draggable = uu.xpath.snap('//div[@class="draggable"]');
145     this.droppable = uu.xpath.snap('//div[@class="droppable"]');
146     this.zindexer = new uu.module.drag.zindexer(); // instantiate
147     // ドラッグ可能要素にマウスカーソルとイベントハンドラを設定する
148     me.hr = uu.event.handler(me);
149     uu.forEach(this.draggable, function(v) {
150       v.style.cursor = me.param.cursor;
151       uu.event.set(v, "mousedown,mousewheel", me.hr);
152     });
153     // ドロップ可能要素の背景色と矩形を独自のプロパティとして保存する
154     uu.forEach(this.droppable, function(v) {
155       v._uu_drag_bgcolor = uu.css.get(v, "backgroundColor");
156       v._uu_drag_rect = uu.ui.element(v);
157     });
158   },
159   /** <b>後処理</b> */
160   destruct: function() {
161     var me = this;
162     uu.forEach(this.draggable, function(v) {
163       uu.event.unset(v, "mousedown,mousewheel", me.hr);
164     });
165   },
166   /** <b>イベントハンドラ</b> */
167   handleEvent: function(evt) {
168     var type = uu.event.type(evt.type);
169     uu.event.stop(evt); // イベントバブルの停止(+テキストの選択を抑止)
170     switch (type) {
171     case "mousedown":
172       uu.event.set(uu.ua.ie ? uu.event.target(evt).real : uud, "mousemove,mouseup", this.hr, true);
173       break;
174     case "mouseup":
175       uu.event.unset(uu.ua.ie ? this.elm : uud, "mousemove,mouseup", this.hr, true);
176       break;;
177     }
178     switch (type) {
179     case "mousedown":
180     case "mousemove":
181     case "mouseup":
182     case "mousewheel":
183       this[type](evt);
184       this.param.msgto && uu.msgpump && this.param.msgFilter(type) &&
185         uu.msgpump.post(this.param.msgto, type, { sender: "uu.module.drag.limited", element: this.elm });
186       break;
187     }
188   },
189   mousedown: function(evt) {
190     this.elm = uu.event.target(evt).real; // ドラッグターゲットの設定
191     var mpos = uu.event.mouse.pos(evt), // マウスの絶対位置を取得
192         rect = uu.ui.element(this.elm); // screen offset
193     // ドラッグ可能要素を絶対座標化
194     uu.ui.element.toAbsolute(this.elm, { opacity: this.param.ghost ? 0.5 : 1.0 });
195     uu.ui.element.toAbsolute(this.elm, { opacity: this.param.ghost ? 0.5 : 1.0 });
196 
197     // z-indexの管理を開始
198     this.zindexer.set(this.elm.id); // z-index初期化
199     this.zindexer.beginDrag(this.elm.id); // z-indexの更新
200     // マウス座標と要素の左上とのオフセットを保存
201     this.param._x = mpos.x - rect.x;
202     this.param._y = mpos.y - rect.y;
203   },
204   mousemove: function(evt) {
205     var mpos = uu.event.mouse.pos(evt);
206     uu.css.set(this.elm, { left: mpos.x - this.param._x + "px",
207                            top:  mpos.y - this.param._y + "px" });
208     this.inDroppableRect(mpos);
209   },
210   mouseup: function(evt) {
211     // ドラッグ可能要素を静的座標化
212     uu.css.set(this.elm, { position: "static", left: 0, top: 0, opacity: 1.0 });
213 
214     // z-indexの管理を終了
215     this.zindexer.endDrag(this.elm.id);
216     this.zindexer.unset(this.elm.id);
217 
218     // droppable要素にドロップ可能かを判断する
219     var e = this.inDroppableRect(uu.event.mouse.pos(evt));
220     if (e) {
221       this.drop(e, this.elm);
222     }
223   },
224   mousewheel: function(evt) {
225     if (!this.param.resize) { return; }
226     var ss = uu.css.get(this.elm), wh = uu.event.mouse.state(evt).wheel * 2;
227     uu.css.set(this.elm, { width:  (parseInt(ss.width)  + (wh * 2)) + "px",
228                            height: (parseInt(ss.height) + (wh * 2)) + "px",
229                            top:    (parseInt(ss.top)    - wh) + "px",
230                            left:   (parseInt(ss.left)   - wh) + "px" });
231   },
232   /** ドロップ時のアクション: */
233   drop: function(droppable, draggable) {
234     // 先客がいるならドロップ失敗
235     if (!uu.xpath('count(./div[@class="draggable"])', droppable)) {
236       if (droppable.hasChildNodes()) {
237         droppable.insertBefore(draggable, droppable.firstChild); // droppable.lastChild
238       } else {
239         droppable.appendChild(draggable);
240       }
241       this.param.msgto && uu.msgpump && this.param.msgFilter("drop") &&
242         uu.msgpump.post(this.param.msgto, "drop", { sender: "uu.module.drag.limited", element: this.elm });
243     }
244   },
245   /** ドロップ可能領域内なら背景色を差し替え: */
246   inDroppableRect: function(mpos) {
247     var rv = null, bg = this.param.dropAllowColor;
248     uu.forEach(this.droppable, function(v) {
249       if (uu.ui.inRect(v._uu_drag_rect, mpos)) {
250         v.style.backgroundColor = bg;
251         rv = v;
252         return;
253       }
254       v.style.backgroundColor = v._uu_drag_bgcolor;
255     });
256     return rv;
257   }
258 };

260 /** <b>ドラッグオブジェクトのz-indexを管理</b>
261  *
262  * @class
263  */
264 uu.module.drag.zindexer = uu.singletonClass();
265 uu.module.drag.zindexer.prototype = {
266   /** <b>初期化</b> */
267   construct: function() {
268     this.obj = {};              // 登録済みのドラッガブルオブジェクト
269     this.boost_zIndex = 1000;   // ドラッグ中のオブジェクトに一時的に設定されるz-index
270     this.default_zIndex = 20;   // 初期オブジェクトの階層z-index
271   },
272   /** <b>IDの登録と適切なz-indexの設定</b>
273    *
274    * @param string id       z-indexを管理する要素のIDを指定します。
275    */
276   set: function(id) {
277     if (id in this.obj) {
278       if (uu.config.debug) { throw Error("duplicate id: " + id); }
279       return;
280     }
281     this.obj[id] = uu.id(id);
282     this.obj[id].style.zIndex = ++this.default_zIndex;
283   },
284   /** <b>IDの抹消</b>
285    * 
286    * @param string id       set()で登録済みのIDを指定します。
287    */
288   unset: function(id) {
289     if (!(id in this.obj)) {
290       if (uu.config.debug) { throw Error("unknown id: " + id); }
291       return;
292     }
293     delete this.obj[id];
294     --this.default_zIndex;
295   },
296   /** <b>ドラッグ開始通知</b>
297    *
298    * @param string id       set()で登録済みのIDを指定します。
299    */
300   beginDrag: function(id) {
301     if (!(id in this.obj)) {
302       if (uu.config.debug) { throw Error("unknown id: " + id); }
303       return;
304     }
305     var base = this.obj[id].style.zIndex;
306     this.obj[id].style.zIndex = this.boost_zIndex + 1;
307 
308     // 毎回ソートするのではなく、
309     // ある値(閾値)を越えた時点でガベージ的にソートを行えば速度向上も可能。
310     uu.forEach(this.obj, function(v) {
311       if (v.style.zIndex > base) { // baseより大きければ-1
312         v.style.zIndex -= 1;
313       }
314     });
315   },
316   /** <b>ドラッグ終了通知</b>
317    *
318    * @param string id       set()で登録済みのIDを指定します。
319    */
320   endDrag: function(id, opacity) {
321     if (!(id in this.obj)) {
322       if (uu.config.debug) { throw Error("unknown id: " + id); }
323       return;
324     }
325     this.obj[id].style.zIndex = this.default_zIndex; // 最上位に移動
326   }
327 };
