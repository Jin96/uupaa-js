 47 /** <b>uu.effect.fade - fade, フェードイン/フェードアウト</b>
 48  *
 49  * 指定された不透明度になるようにアニメーションします。<br />
 50  * beginとendを省略すると、現在の不透明度が0.5以上ならfadeoutし、0.5未満ならfadeinします。<br />
 51  * 要素が非表示になっている場合は、まず可視状態にしてからアニメーションを行います。
 52  *
 53  * @param element         elm           - 要素を指定します。
 54  * @param hash            [param]       - パラメタの指定です。
 55  * @param number/string   [param.speed] - 描画完了までの時間を、ms単位の数値か、
 56  *                                        文字列("quick", "mid", "slow")で指定します。デフォルトは"mid"です。
 57  * @param function        [param.fn]    - エフェクト完了でコールバックするメソッドを指定します。省略可能です。
 58  * @param number          [param.begin] - 初期不透明度を指定します。デフォルトは現在の不透明度です。
 59  * @param number          [param.end]   - 目標とする不透明度を指定します。デフォルトは0.0(完全な透明)または1.0(完全な不透明)です。
 60  */
 61 uu.effect.fade = function(elm, param /* = { speed: "mid", fn: undefined, begin: current-opacity, end: undefined } */) {
 62   var curt = uu.css.get.opacity(elm),
 63       pa = uu.mix.param(param || {}, { speed: "mid", fn: uu.mute, begin: -1, end: -1 });
 64   if (pa.begin === -1 || !uu.effect._judgeOpacity(pa.begin)) { pa.begin = curt; }
 65   if (pa.end   === -1 || !uu.effect._judgeOpacity(pa.end))   { pa.end = (curt < 0.5) ? 1.0 : 0.0; }
 66   uu.effect.show(elm);
 67   (pa.begin > pa.end) ? uu.effect.fadeout._impl(elm, pa.speed, pa.fn, pa.begin, pa.end) :
 68                         uu.effect.fadein._impl(elm, pa.speed, pa.fn, pa.begin, pa.end);
 69 };

 71 /** <b>uu.effect.fadein - fadein, フェードイン</b>
 72  *
 73  * 指定された不透明度になるようにアニメーションします。<br />
 74  * beginとendを省略すると、完全に透明な状態から完全に不透明な状態にアニメーションします。<br />
 75  * 要素が非表示になっている場合は、まず可視状態にしてからアニメーションを行います。
 76  *
 77  * @param element         elm           - 要素を指定します。
 78  * @param hash            [param]       - パラメタの指定です。
 79  * @param number/string   [param.speed] - 描画完了までの時間を、ms単位の数値か、
 80  *                                        文字列("quick", "mid", "slow")で指定します。デフォルトは"mid"です。
 81  * @param function        [param.fn]    - エフェクト完了でコールバックするメソッドを指定します。省略可能です。
 82  * @param number          [param.begin] - 初期不透明度を指定します。デフォルトは現在の不透明度です。
 83  * @param number          [param.end]   - 目標とする不透明度を指定します。デフォルトは1.0(完全な不透明)です。
 84  */
 85 uu.effect.fadein = function(elm, param /* = { speed: "quick", fn: undefined, begin: current-opacity, end: 1.0 } */) {
 86   var curt = uu.css.get.opacity(elm),
 87       pa = uu.mix.param(param || {}, { speed: "mid", fn: uu.mute, begin: curt, end: 1.0 });
 88   uu.effect.show(elm);
 89   uu.effect.fadein._impl(elm, pa.speed || 1, pa.fn, uu.effect._judgeOpacity(pa.begin) ? pa.begin : 0.0,
 90                                                     uu.effect._judgeOpacity(pa.end)   ? pa.end   : 1.0);
 91 }
 92 uu.effect.fadein._impl = function(elm, speed, fn, begin, end) {
 93   var me = uu.effect.fadein, delta, curt = begin;
 94   (!uu.isN(speed)) && (speed = me._speed[(speed in me._speed) ? speed : "mid"]);
 95   delta = 1 / (speed / 10);
 96 
 97   function loop(step) {
 98     switch (step) {
 99     case 1: if (curt >= end) { return false; }
100             break;
101     case 2: uu.css.set.opacity(elm, ((curt += delta) > end) ? (curt = end) : curt);
102             break;
103     case 4: fn();
104     }
105     return true;
106   }
107   uu.effect._frame(elm, 10, loop);
108 };
109 uu.effect.fadein._speed = { quick: 250, mid: 1000, slow: 4000 };

112  *
113  * 指定された不透明度になるようにアニメーションします。<br />
114  * beginとendを省略すると、完全に不透明な状態から完全に透明な状態にアニメーションします。<br />
115  * 要素が非表示になっている場合は、まず可視状態にしてからアニメーションを行います。
116  *
117  * @param element         elm           - 要素を指定します。
118  * @param hash            [param]       - パラメタの指定です。
119  * @param number/string   [param.speed] - 描画完了までの時間を、ms単位の数値か、
120  *                                        文字列("quick", "mid", "slow")で指定します。デフォルトは"mid"です。
121  * @param function        [param.fn]    - エフェクト完了でコールバックするメソッドを指定します。省略可能です。
122  * @param number          [param.begin] - 初期不透明度を指定します。デフォルトは現在の不透明度です。
123  * @param number          [param.end]   - 目標とする不透明度を指定します。デフォルトは0.0(完全な透明)です。
124  */
125 uu.effect.fadeout = function(elm, param /* = { speed: "quick", fn: undefined, begin: current-opacity, end: 0.0 } */) {
126   var curt = uu.css.get.opacity(elm),
127       pa = uu.mix.param(param || {}, { speed: "mid", fn: uu.mute, begin: curt, end: 0.0 });
128   uu.effect.show(elm);
129   uu.effect.fadeout._impl(elm, pa.speed || 1, pa.fn, uu.effect._judgeOpacity(pa.begin) ? pa.begin : 1.0,
130                                                      uu.effect._judgeOpacity(pa.end)   ? pa.end   : 0.0);
131 }
132 uu.effect.fadeout._impl = function(elm, speed, fn, begin, end) {
133   var me = uu.effect.fadein, delta, curt = begin;
134   (!uu.isN(speed)) && (speed = me._speed[(speed in me._speed) ? speed : "mid"]);
135   delta = 1 / (speed / 10);
136 
137   function loop(step) {
138     switch (step) {
139     case 1: if (curt <= end) { return false; }
140             break;
141     case 2: uu.css.set.opacity(elm, ((curt -= delta) < end) ? (curt = end) : curt);
142             break;
143     case 4: fn();
144     }
145     return true;
146   }
147   uu.effect._frame(elm, 10, loop);
148 };

151 /** <b>uu.effect._frame - エフェクトフレーム</b>
152  *
153  * エフェクトはbeginからendまでの各ステップで構成されており、beforeとafterの間は無限ループします。<br /><br />
154  *
155  * <pre>
156  * begin(0)　→　before(1)　→　effect(2)　→　after(3)　→　end(4)
157  * 　　　　　　　　　↑　　　　　　　　　　　　　│
158  * 　　　　　　　　　└─────────────┘
159  * </pre>
160  *  <table>
161  *  <tr><th>step番号</th><th>内容</th></tr>
162  *  <tr><td>0</td><td>BEGIN: 処理開始       </td></tr>
163  *  <tr><td>1</td><td>BEFORE: effect実行前  </td></tr>
164  *  <tr><td>2</td><td>EFFECT: effect本処理  </td></tr>
165  *  <tr><td>3</td><td>AFTER: effect実行後   </td></tr>
166  *  <tr><td>4</td><td>END: 処理終了         </td></tr>
167  *  </table>
168  * 各ステップで、fn(step番号)を呼び出します。<br />
169  * before,effect,afterの各ステップでfalseを返すとendに移行します。<br />
170  *
171  * @param element         elm     - 要素を指定します。
172  * @param number          speed   - 速度の指定です。単位はmsです。
173  * @param function        fn      - コールバック関数を指定します。
174  */
175 uu.effect._frame = function(elm, delay, fn) {
176   var run = 0, vtid = 0;
177   function finish() { !run++ && (uu.effect._vtm.suspend(vtid), fn(4));         }
178   function loop()   { (!run && fn(1) && fn(2) && fn(3)) ? 0 : finish(); }
179   fn(0);
180   vtid = uu.effect._vtm.set(loop, delay);
181 };

183 /** uu.effect._judgeOpacity - 不透明度の検査 */
184 uu.effect._judgeOpacity = function(opacity) {
185   return (uu.isN(opacity) && opacity >= 0.0 && opacity <= 1.0);
186 }
