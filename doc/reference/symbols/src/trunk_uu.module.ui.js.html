133 /** <b>最寄の基準点からの累計オフセットと要素の大きさを取得</b>
134  *
135  * 最寄の基準点からの累計オフセットと要素の大きさを取得します。
136  * 最寄の基準点とは、position absoluteやrelativeが指定された要素のことです。
137  *
138  *  <dl>
139  *    <dt>x</dt><dd>最寄の基準点からの累計オフセット値</dd>
140  *    <dt>y</dt><dd>最寄の基準点からの累計オフセット値</dd>
141  *    <dt>w</dt><dd>要素の幅(オフセット幅, 左右のボーダーを含む) - offsetWidth</dd>
142  *    <dt>h</dt><dd>要素の高さ(オフセット高さ, 上下のボーダーを含む) - offsetHeight</dd>
143  *    <dt>cw</dt><dd>要素の幅(オフセット幅, 左右のボーダーを含まない) - clientWidth</dd>
144  *    <dt>ch</dt><dd>要素の高さ(オフセット高さ, 上下のボーダーを含まない) - clientHeight</dd>
145  *  </dl>
146  * @param Element elm - 要素を指定します。
147  * @return rect       - { x, y, w, h, cw, ch } を返します。<br />
148  */
149 uu.ui.element.offsetParent = function(elm) {
150   var e = elm, x = 0, y = 0, ss;
151   while (e) {
152     x += e.offsetLeft || 0;
153     y += e.offsetTop  || 0;
154     e = e.offsetParent;
155     if (e) {
156       ss = uu.css.get(e, "position"); // position を持つものが基準点となる
157       if (ss === "relative" || ss === "absolute") { break; }
158     }
159   }
160   return { x: x, y: y,
161            w: elm.offsetWidth, h: elm.offsetHeight,
162            cw: elm.clientWidth, ch: elm.clientHeight };
163 };

165 /** <b>絶対座標化 - Absolute positioning</b>
166  *
167  * 要素の現在位置を保持したまま絶対座標化します。
168  *
169  * @param   element   elm     - 要素を指定します。
170  * @param   hash      [style] - 追加するスタイルを { 名前: 値, ... }の形式で指定します。<br />
171  *                              デフォルトはundefinedです。
172  * @return  element           - elmを返します。
173  */
174 uu.ui.element.toAbsolute = function(elm, style /* = undefined */) {
175   var rect = uu.ui.element(elm);
176   style = uu.mix.param(style || {}, { position: "absolute", left: rect.x + "px", top: rect.y + "px" });
177   uu.css.set(elm, style);
178   return elm;
179 };

195 /** <b>座標が矩形内にあればtrue</b>
196  *
197  * 座標(マウスカーソル等)が矩形内に含まれている場合にtrueを返します。
198  *
199  * <pre>
200  *  rect━┯━━┓
201  *    ┃  │    ┃
202  *    ┠─pos   ┃
203  *    ┃        ┃
204  *    ┗━━━━┛
205  * </pre>
206  *
207  * @param   rect    rect - { x, y, w, h } 要素の座標と大きさを指定します。
208  * @param   pos     pos  - { x, y } 点を指定します。
209  * @return  bool         - rect上にposがあればtrueを返します。
210  */
211 uu.ui.inRect = function(rect, pos) {
212   var r = rect, p = pos;
213   return (p.x > r.x && p.x < r.x + r.w) && (p.y > r.y && p.y < r.y + r.h);
214 };

307 /** <b>マウスクリック, ホイールの状態を取得</b>
308  *
309  * 以下の状態を含むhashを返します。<br />
310  * left:  左クリックでtrue<br />
311  * mid:   中クリックでtrue<br />
312  * right: 右クリックでtrue<br />
313  * click: クリック数, シングルクリックで1, ダブルクリックで2, トリプルクリックで3<br />
314  * wheel: マウスホイールを上に回転させると-1を、下に回転させると1を返します<br />
315  *
316  * @param   event   evt - イベントオブジェクト
317  * @return  hash        - { left, mid, right, click, wheel } を返します。
318  */
319 uu.event.mouse.state = function(evt) {
320   var rv = { left: 0, mid: 0, right: 0, click: 0, wheel: 0 };
321   if (uu.ua.gecko || evt.which) {
322     switch (evt.which) {
323       case  1: rv.left  = 1; break;
324       case  2: rv.mid   = 1; break;
325       case  3: rv.right = 1; break;
326     }
327     rv.click = evt.detail & 0x03; // クリック回数(0～3)
328   } else if (uu.ua.ie || evt.button) {
329     if (evt.button & 0x1) { rv.left = 1;  }
330     if (evt.button & 0x4) { rv.mid = 1;   }
331     if (evt.button & 0x2) { rv.right = 1; }
332   }
333   // wheelDelta: Safari, IE
334   // detail: Firefox
335   if (evt.wheelDelta || evt.detail) {
336     rv.wheel = parseInt(evt.detail ? (evt.detail / 3) : (evt.wheelDelta / -120));
337   }
338   return rv;
339 };

357 /** <b>uu.event.mouse.clicks - クリックイベントハンドラの設定 - Set click event handler</b>
358  *
359  * マウスクリックをトリガーに動作する関数を登録します。<br />
360  * fnはいくつでも指定可能で、複数の関数を登録すると、turn around(toggle) 動作を行います。
361  *
362  * @param Element elm  - イベントを設定する要素を指定します。
363  * @param Function fn  - クリック時に呼ばれる関数を指定します。
364  */
365 uu.event.clicks = function(elm, fn, toggles /* = 2 */) {
366   elm.uuClicks = { idx: 0, max: toggles || 2, fn: fn };
367   uu.event.set(elm, "click", uu.event.clicks._impl);
368 };
369 uu.event.clicks._impl = function(evt) {
370   (uu.ua.ie) ? (evt.returnValue = false) : evt.preventDefault();
371   var v = uu.event.target(evt).real.uuClicks;
372   v.fn((++v.idx >= v.max) ? (v.idx = 0) : v.idx);
373 };
