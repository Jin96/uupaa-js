 59 /** 発声
 60  * @param String text - テキストを指定します。
 61  * @param Number [delay] - 遅延時間を指定します。単位はmsです。デフォルトは0です。

 63 uu.perry.say = function(text, delay) {
 64   if (uu.perry._delayTimerID !== -1) {
 65     clearTimeout(uu.perry._delayTimerID);
 66     uu.perry._delayTimerID = -1;
 67   }
 68   if (!uu.isS(text) || !text.length) { return; }
 69   uu.perry._delayTimerID = uuw.setTimeout(function() {
 70     uu.perry._say(text);
 71   }, delay);

 73 uu.perry._say = function(text) {
 74   var rv = [], i, sz, v1, v2, v3, v4, v5, voice = uu.perry.dict.voice,
 75       ascii = 0;
 76   if (!uu.perry._obj) {
 77     try {
 78     uu.perry._obj = (uu.ua.ie)    ? new ActiveXObject("SAPI.SpVoice") :
 79                     (uu.ua.gecko) ? new GeckoActiveXObject("SAPI.SpVoice") : null;
 80     } catch (e) {
 81       throw Error("uu.perry.say(create ActiveXObject(\"SAPI.SpVoice\"))");
 82     }
 83   }
 84   if (!uu.perry._obj) { throw Error("uu.perry.say(create ActiveXObject(\"SAPI.SpVoice\"))"); }
 85 
 86   // 各メソッドで前処理
 87   for (i = 0, sz = uu.perry.method._chain.length; i < sz; ++i) {
 88     text = uu.perry.method._chain[i](text);
 89   }
 90   // カナ/ひらがな混じりのテキストを発声記号に変換
 91   text = (text + "     ").split("");
 92   for (i = 0, sz = text.length; i < sz; ++i) {
 93     v1 = text[i];
 94     if (v1 === " " || v1.match(/[a-zA-Z0-9\-_,\.\?]/)) { // スペースとアルファベットは変換しない
 95       if (!ascii) { // "のOSが" → "noOS ga" にならないよう、単語の区切りでスペースを挿入する
 96         rv.push(" ");
 97       }
 98       rv.push(v1);
 99       ascii = 1;
100     } else {
101       ascii = 0;
102       // カナ(kana) → ひらがな(hiragana)
103       v1 = kanaToHiragana(v1);
104       v2 = v1 + kanaToHiragana(text[i + 1]);
105       v3 = v2 + kanaToHiragana(text[i + 2]);
106       v4 = v3 + kanaToHiragana(text[i + 3]);
107       v5 = v4 + kanaToHiragana(text[i + 4]);
108       if (v5 in voice) { rv.push(" ", voice[v5]); i += 4; } else
109       if (v4 in voice) { rv.push(" ", voice[v4]); i += 3; } else
110       if (v3 in voice) { rv.push(" ", voice[v3]); i += 2; } else
111       if (v2 in voice) { rv.push(" ", voice[v2]); i += 1; } else
112       if (v1 in voice) { rv.push(" ", voice[v1]); }
113     }
114   }
115   // ドット(".")で段落を区切り、ブラウザの無反応時間を短くする。
116   rv.join("").split(".").forEach(function(v) {
117     window.status = v;
118     uu.perry._obj.Speak(v);
119   });
120 
121   function kanaToHiragana(str) {
122     var hira = "ぁ".charCodeAt(0),
123         p1 = "ァ".charCodeAt(0),
124         p2 = "ヶ".charCodeAt(0),
125         c = str.charCodeAt(0);
126     return (c >= p1 && c <= p2) ? String.fromCharCode(c - (p1 - hira)) : str;
127   }

206   var rv = [], i = 0, sz,

227     }

275       case  8: rv.push("おく"); break;
276       case 12: rv.push("ちょう"); break;
277       case 16: rv.push("けい"); break;

454  *
455  * - 促音(a double [long] consonant)
456  * -- [きる][て] → きって
457  * -- [かく][かん] → らっかん
458  * -- [もく][きん] → もっきん
459  * -- [とく][くん] → とっくん
460  * -- [とく][けん] → とっけん
461  * -- [がく][こう] → がっこう
462  * -- [けつ][こん] → けっこん
463  * -- [まつたく]   → まったく
464  * -- [いつつい]   → いっつい
465  * -- [かつ][とう] → かっとう
466  * -- [くつ][つく] → くっつく
467  * -- [さつち]     → さっち
468  * -- [いち][はん] → いっぱん
469  * -- [さつ][ち]   → さっち
470  * -- ニチ + キ = ニッキ
471  * -- サク + カ = サッカ
472  * -- ニチ + ホン = ニッポン (これは例外か?)
473  * --- 変換ルール   {は行} + 子音が"つ" + {は行} → {は行} + "っ" + {は行} + 半濁音
474  * --- ハツ + ヒョウ + シャ = ハッピョウシャ
475  *
476  * - わかち書き
477  * -- 格助詞で分断:   の, に, を, へ, と, から, より, で, が
478  * -- 並立助詞で分断: の, に, やら, か, なり, だの,
479  * -- 終助詞で分断:   の, か, かしら, とも, な, ぞ, や, わ
480  * -- 間投助詞で分断: さ, よ, ね
481  * -- 副助詞で分断:   ばかり, まで, だけ, ほど, くらい, など, なり, やら
