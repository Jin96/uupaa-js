 10 /** <b>Image</b>
 11  *
 12  * @class

 14 uu.module.image = function() {
 15 };

 17 /** <b>画像のロード済み確認 - Is image loaded</b>
 18  *
 19  * 画像をロード済み(すぐに使用可能な状態)ならtrueを返します。
 20  *
 21  * @param String url - 画像のURLを指定します。絶対URLを指定します。
 22  * @return Boolean   - ロード済みでtrueを返します。
 23  */
 24 uu.module.image.isLoaded = function(url) {
 25   function judge(node) { return node.complete && node.src === url; }
 26   return uu.toArray(uud.images).some(judge);
 27 };
 28 
 29 /** <b>画像のプリロード - Preload image</b>
 30  *
 31  * 画像を非同期にロードします。
 32  * リクエスト終了でfn(code, url, image要素)をコールします。<br />
 33  * codeには成功で1,失敗で0が格納されます。<br />
 34  *
 35  * このメソッドはロールオーバーやcanvasで使用する画像のプリロード用です。<br /><br />
 36  *
 37  * 既に画像が使用可能になっている状態(uu.module.image.isLoaded()==true)で呼ばれた場合は、
 38  * 即座に関数を実行します。
 39  *
 40  * @param String   url - 画像のパスの指定です。絶対URLや相対パスを指定します。
 41  * @param Function fn  - リクエスト終了で呼び出す関数を指定します。デフォルトはuu.muteです。
 42  * @see <a href="#timeout">uu.module.image.timeout</a> - タイムアウト時間の指定
 43  * @see <a href="#delay">uu.module.image.delay</a> - 遅延時間の指定
 44  * @see <a href="http://d.hatena.ne.jp/uupaa/20080413/1208067631">uupaa開発日記</a>
 45  */
 46 uu.module.image.preload = function(url, fn /* = uu.mute */) { // for Firefox3, Safari3.1, Opera9.5β3, IE6/7/8β
 47   var tick = 0, img, run = 0; // 二重起動防止用
 48 
 49 //url = url.toAbsURL();
 50   url = uu.url.abs(url);
 51   fn = fn || uu.mute;
 52   img = new Image();
 53 
 54   if (uu.module.image.isLoaded(url)) { // already loaded
 55     img.src = url;
 56     fn(1, url, img);
 57     return;
 58   }
 59   img.onabort = img.onerror = function() { run++ ? 0 : fn(0, url, img); };
 60   img.onload  = function() {
 61     run++ ? 0 : ((uu.ua.opera && !img.complete) ? fn(0, url, img) : fn(1, url, img));
 62   };
 63   img.src = url;
 64 
 65   if (!run && uu.module.image.timeout) {
 66     uuw.setTimeout(function() {
 67       if (run) { return; }
 68       if ((uu.ua.gecko && img.complete && !img.width) || // Firefox2は読込失敗でcomplete=true, width=0になる
 69           (tick += uu.module.image.delay) > uu.module.image.timeout) { // タイムアウト
 70         run++ ? 0 : fn(0, url, img);
 71         return;
 72       }
 73       uuw.setTimeout(arguments.callee, uu.module.image.delay);
 74     }, 0);
 75   }
 76 };
 77 
 78 /** <b>タイムアウト時間 - Timeout</b>
 79  * uu.module.image.load()で使用するタイムアウト時間を指定します。
 80  * 1以上の数値を指定します。単位はmsです。
 81  * デフォルトは10000です。
 82  *
 83  * @type number
 84  * @see <a href="#preload">uu.module.image.preload()</a> - 画像のプリロード
 85  */
 86 uu.module.image.timeout = 10000; // 10000ms(10s)
 87 
 88 /** <b>遅延時間 - Delay</b>
 89  * uu.module.image.load()で使用する遅延時間の指定です。
 90  * 10以上の数値を指定します。単位はmsです。デフォルトは50です。
 91  *
 92  * @type number
 93  * @see <a href="#preload">uu.module.image.preload()</a> - 画像のプリロード
 94  */
 95 uu.module.image.delay = 50; // 50ms
 96 
 97 })(); // end (function())()
