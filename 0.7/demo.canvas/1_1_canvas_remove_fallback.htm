function canvasReady() {
    lazyLoad("../uupaa.js", function() {
        uu.canvas.init();
        drawShape(document.getElementById('slcanvas'));
        drawShape(document.getElementById('vmlcanvas'));
        drawShape(document.getElementById('flashcanvas'));
    });
}

function lazyLoad(url, callback) {
    var scr     = document.createElement("script");
    scr.type    = "text/javascript";
    scr.charset = "utf-8";

    if (document.uniqueID) {
        scr.onreadystatechange = function() {
            if (/loaded|complete/.test(this.readyState)) {
                  callback();
            }
        };
    } else {
        scr.onload = function() {
            callback();
        };
    }

    scr.setAttribute("src", url);
    document.getElementsByTagName("head")[0].appendChild(scr);
}

function drawShape(canvas) {
    if (canvas.getContext) { return; }

    var ctx = canvas.getContext('2d');

    ctx.fillRect(25,25,100,100);
    ctx.clearRect(45,45,60,60);
    ctx.strokeRect(50,50,50,50);
}
