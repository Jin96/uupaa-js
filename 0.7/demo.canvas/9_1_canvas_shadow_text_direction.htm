A canvas shadow text direction example



      <canvas id="slcanvas" class="sl" width="360" height="400"></canvas>
      <canvas id="vmlcanvas" class="vml" width="360" height="400"></canvas>
      <canvas id="flashcanvas" class="flash" width="360" height="400"></canvas>

function xcanvas() {
    drawShape(document.getElementById('slcanvas'));
    drawShape(document.getElementById('vmlcanvas'));
    drawShape(document.getElementById('flashcanvas'));
}
function matrix() {
    drawShape(document.getElementById('slcanvas'), 1);
    drawShape(document.getElementById('vmlcanvas'), 1);
    drawShape(document.getElementById('flashcanvas'), 1);
}

var globalAlpha = 1.0;
function alpha() {
    globalAlpha -= 0.2;

    drawShape(document.getElementById('slcanvas'));
    drawShape(document.getElementById('vmlcanvas'));
    drawShape(document.getElementById('flashcanvas'));
}

function drawShape(canvas, matrix){
    if (!canvas.getContext){ return; }

    var ctx = canvas.getContext('2d');

    ctx.globalAlpha = globalAlpha;

    ctx.clear();
    if (matrix) {
        ctx.translate(-10, 5);
        ctx.scale(1.2, 0.8);
        ctx.rotate(5 * Math.PI / 180);
    }
    ctx.textBaseline = "top";
    ctx.font = "24pt Arial";
    ctx.fillStyle = "tomato";

    var text;

    text = setShadow(ctx, "blue", 0, 0, 0);
    ctx.fillText("fillText: "       + text, 20, 20);
    text = setShadow(ctx, "blue", 10, 0, 1);
    ctx.fillText("fillText: "       + text, 20, 60);
    text = setShadow(ctx, "blue", 10, -10, 2);
    ctx.fillText("fillText: "       + text, 20, 100);
    text = setShadow(ctx, "blue", 0, -10, 3);
    ctx.fillText("fillText: "       + text, 20, 140);
    text = setShadow(ctx, "blue", -10, -10, 4);
    ctx.fillText("fillText: "       + text, 20, 180);
    text = setShadow(ctx, "blue", -10, 0, 5);
    ctx.fillText("fillText: "       + text, 20, 220);
    text = setShadow(ctx, "blue", -10, 10, 6);
    ctx.fillText("fillText: "       + text, 20, 260);
    text = setShadow(ctx, "blue", 0, 10, 7);
    ctx.fillText("fillText: "       + text, 20, 300);
    text = setShadow(ctx, "blue", 10, 10, 8);
    ctx.fillText("fillText: "       + text, 20, 340);
}
function setShadow(ctx, color, ox, oy, blur) {
    ctx.shadowColor = color;
    ctx.shadowOffsetX = ox;
    ctx.shadowOffsetY = oy
    ctx.shadowBlur = blur;
    return [color, ox, oy, blur].join(", ");
}
